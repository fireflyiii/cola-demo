@UtilityClass
public final class PageParamConverter {

    public static <T> IPage<T> pageParamToPage(BaseReqParam query) {

        FrameworkExceptionEnum.PAGE_PARAM_NOT_NULL.assertNotNull(query);
        FrameworkExceptionEnum.PAGE_PARAM_NOT_NULL.assertNotNull(query.getPageParam());

        PageParam pageParam = query.getPageParam();

        return new Page<T>().setPages(pageParam.getPages()).setCurrent(pageParam.getCurrentPage()).setSize(pageParam.getPageSize()).setTotal(pageParam.getTotalNum());

    }

    public static PageParam pageToPageParam(@SuppressWarnings("rawtypes") IPage data) {

        FrameworkExceptionEnum.PAGE_PARAM_NOT_NULL.assertNotNull(data);
        return new PageParam().setCurrentPage(data.getCurrent()).setPages(data.getPages()).setTotalNum(data.getTotal()).setPageSize(data.getSize());

    }
